Общая структура проекта:

├── src/
│   ├── converters/

│   │   ├── algorithmic_indexing/
│   │   │   ├── correction_functions.py  	# Корректура заголовков
│   │   │   ├── add_categories.py     		# Подбор рубрик
│   │   │   └── edd_keywords.py      		# Подбор ключевых слов
│   │   ├── authors_functions.py		# Обработка авторов
│   │   ├── list_to_dict.py			# Преобразование из списка в словарь 
│   │   └── multifile_import.py     		# Пакетная загрузка HTML
│   ├── web/
│   │   ├── __init__.py
│   │   ├── app.py                  		# Основное Flask‑приложение
│   │   ├── routes/
│   │   │   ├── converter_routes.py 		# Маршрутизация конвертеров
│   │   │   ├── api_routes.py      		# API‑эндпоинты
│   │   │   └── __init__.py
│   │   ├── templates/
│   │   │   ├── base.html           		# Базовый шаблон
│   │   │   ├── index.html         		# Главная страница
│   │   │   ├── upload.html       		# Загрузка файлов
│   │   │   └── result.html      		# Результат конвертации
│   │   └── static/
│   │       ├── css/
│   │       │   └── style.css
│   │       └── js/
│   │           └── main.js
│   └── utils/
│       ├── file_manager.py         		# Работа с файловой системой
│       ├── config_loader.py       		# Загрузка конфигураций
│       └── logger.py             		# Логирование
├── data/
│   ├── config/
│   │   ├── yandex_cloud_config.json  		# Токен Yandex Cloud
│   │   └── path_config.json   			# Пути к файлам проекта
│   ├── databases/
│   │   ├── a_facet.xlsx            		# Имена собственные для корректуры
│   │   ├── category_rubricator.json		# Рубрикаторы
│   │   ├── descriptors.json        		# Нормализованные ключевые слова
│   │   ├── descriptors_categories.xlsx		# Соответствие рубрик дескрипторам
│   │   ├── journal_list.json       		# База журналов
│   │   └── persons.json            		# Нормализованные персоналии
│   └── templates/
│       └── default_template.json  		# Шаблон по умолчанию
├── resources/
│   ├── files_to_process/			# Папка для исходных файлов в html
│   ├── files_to_edit/				# Папка для промежуточных таблиц
│   └── files_to_import_to_IRBIS/		# Папка для файлов в формате ИРБИС
├── tests/
│   ├── test_converters/
│   │   ├── test_elibrary_excel.py		# Тестовый вариант конвертера elibrary-excel
│   │   └── test_excel_irbis.py			# Тестовый вариант конвертера excel-irbis
│   ├── test_services/
│   │   ├── test_translation.py			# Тестовый переводчик
│   │   └── test_correction.py			# Тестовые функции корректуры
│   └── conftest.py
├── .env                    			# Переменные окружения
├── requirements.txt      			# Зависимости
├── main.py               			# Точка входа (CLI)
└── run.py                			# Запуск Flask‑приложения
